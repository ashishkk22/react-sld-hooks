# useRenderCount

The useRenderCount hook is a custom React hook that enables you to get the total render of any component.

### Reference

```ts
const useRenderCount = () => number;
```

### Usage

```tsx
import { useRenderCount } from "react-sld-hooks";

const Example: React.FC<ExampleProps> = ({ initialCounter }) => {
  const [counter, setCounter] = React.useState(initialCounter);

  const renderCount = useRenderCount();

  return (
    <div>
      <div>current counter : {counter}</div>
      <button onClick={() => setCounter((prev) => prev + 1)}>increase</button>
      <div>component render count : {renderCount}</div>
    </div>
  );
};
```
